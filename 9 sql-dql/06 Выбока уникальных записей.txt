
	Выборка в которой отсутствуют дубли.
					select distinct name from products; 	// После distinct  указываем поле в котором будут убираться дубли
					
	Иногда нужно ублрать дубли из нескольких полей (смотрится уникальность пар)
					select distinct price, name from products;		// Происходит частично дублирование (т.е. могут быть два  bmw  но с разным price)
					
	Если нужно и по парно и поодному То:
					select distinct on (name) name, price from products;	// НЕ гарантируется порядок вывода (даже что price соответсвует  name)
	Обязательно  order by				
					select distinct on (name) name, price from products order by name;
