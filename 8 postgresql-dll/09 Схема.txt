
				\d 		// просмотр описания объектов в текуще БД (показывает какие таблицы)
						// Там будут поля Name (имя таблицы) Owner (пользователь) Type (типы: table или sequnce (см. урок 7))
						// также будет Schema  (со значениями  public ..)
						
	Схема это возможность стандарта SQL, позволяющая объединять объекты БД в логические группы.
		Например есть пользователи и Вы не хотите, чтобы они пересекались друг с другом.
		Например есть сторонние приложения, которые работают в БД и вы хотите сделать их независимыми.
	
	Существует несколько причин, почему вы можете захотеть использовать схемы:
		Чтобы разрешить нескольким пользователям использовать одну базу данных, не смешивая их данные.
		Чтобы организовать объекты базы данных в логические группы, для того чтобы сделать их более прозрачными в управлении.
		Разные приложения могут быть размещены в отдельных схемах, чтобы не было конфликтов в использовании имён или других объектов.	
	
	
	По умолчанию в PSQL  все таблицы попадают в схему  public.
	
						CREATE SCHEMA mysuperschema;		// Создание схемы с именем mysuperschema.
						\? 									// Вызов справки по командам
						\dn									// Список всех схем в БД
						\dt *.*								// Список всех таблиц во всех схемах и таблицах текущей БД
						DROP SCHEMA mysuperschema;			// Удаление схемы 
						DROP SCHEMA mysuperschema cascade;	// Удаление схемы  даже если в ней есть таблицы и другие сущности
						
						
	information_schema	   содержит  вью  представления (в которых отражены возможности нашей БД: на каких языках можно писать в нашей БД и т.д.)
	pg_catalog			  все остальные объекты PSQL которые по сути встроены в систему БД (функции и т.д.)
	
	при запросе какой либо таблицы   поиск идет  через  search_parth
	
	Полное название 
						CREATE TABLE myschema.mytable ( ... );		// Создание табл в опред схеме
						SELECT * FROM schema.table ....				// Выборка
						
						база.схема.таблица							// еще более полное название
						
	Если схему не указываем, то в PSQL  по умолчанию все попадает в public.
