ЗАДАНИЕ
	/src/App/Solution/Solution.php
	Реализуйте функцию solution которая принимает на вход список чисел и выполняет следующие действия:

	округляет все числа в списке до верхней границы.
	удаляет нечетные числа.
	возвращает произведение оставшихся элементов.
	
	Подсказки
	Округление до верхней границы: ceil.
	Используйте функции map, filter, reduce из пакета hexlet/pairs, определение которых можно 
	 подсмотреть тут https://github.com/hexlet-components/php-pairs.
	 
	 
	 
								<?php

								namespace App\Solution;

								use function Lists\map;
								use function Lists\filter;
								use function Lists\reduce;

								// BEGIN (write your solution here)

								function solution($list)
								{
									$r1 = map(function ($item) {
										return ceil($item);
									}, $list);

									$r2 = filter(function ($item) {
										return $item % 2 === 0;
									}, $r1);

									$r3 = reduce(function ($item, $acc) {
										return $acc * $item;
									}, $r2, 1);

									return $r3;
								}

								// END
