
	Константы испльзуются когда-есть некоторое значение не меняющееся со временем.
			
		Описание	
			define ('VERSION', "version-1.5");		// это  global definition (глобальная не привязана к namespace).
			const FOO = 7 ** 10;					// более новый способ 
		define - функция, const - инструкция.
		define определяет глобальную константу вне неймспейсов, 
		const определяет константу в текущем неймспейсе.
			
			
		Использование (так называемая подстановочная модель).
			echo VERSION;	
			echo FOO;
			
		
			defined('\Theory\VERSION'); 		// Проверка константы на существование (проверяем по пространству имен).
			constant('FOO');					// Принимает имя как строчку и ищет в своих таблицах, возвращая значение.
				
	Предоопределенные константы.
	Predefined Constants.
		PHP_VERSION;
		PHP_OS;
		PHP_BINDIR;
		DEFAULT_INCLUDE_PATH;
		
	Магические константы.
	Magic Constants.   В реальности значения этих констант зависит от места, где они вызваны.
		__LINE__;	// текущая строчка
		__FILE__;	//
		__DIR__;	// путь до текущ директории
		
		__FUNCTION__, __CLASS__, __METHOD__, __TRAIT__, __NAMESPACE__
