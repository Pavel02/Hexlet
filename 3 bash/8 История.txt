
	История команд сохраняет в файле в домашней директории
			
			cat ~/.bash_history		// Содержимое файла в домашней директории.
			
			history 				// Вся история
			
			!522					// Отработает команда 
			
			!!						// Запуск последней в истории команды
			
			ctrl + r				// Поиск по всей истории команд (поиск далее ctrl + R)
