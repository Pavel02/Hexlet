
	Команды изменяющие файловую структуру
		
			mkdir name				// Создает директорию name
			mkdir -p test2/test3	// Создадутся 2 каталога
			
			touch file1				// Создание файла с именем file1
			touch test/file2		// Создание файла с именем file2 в каталоге test
			
	В Unix системах как такового переименования не существует, а существует перемещение.
			mv test test3			// Переименуется каталог  test   в  test3
			
			mv test2/* test3		// Перемещение всех файлов из каталога test2  в каталог  test3
			
			rm file1				// Удаление файла  file1
			
			rm -r test1				// Удаление директории  (по умолчанию не даст нужен ключ  -r)
			
			rm -r *					// Удаление всех файлов внутри текущей директории
			
			rm -rf test2/*			// Удалить все файлы в каталоге  test2
